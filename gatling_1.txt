package utils;

public class EnvironmentConfig {

    private static String env = System.getProperty("env", "qa");

    public static String authUrl() {
        switch (env) {
            case "uat": return "https://uat-auth.company.com";
            case "prod": return "https://auth.company.com";
            default: return "https://qa-auth.company.com";
        }
    }

    public static String userUrl() {
        switch (env) {
            case "uat": return "https://uat-user.company.com";
            case "prod": return "https://user.company.com";
            default: return "https://qa-user.company.com";
        }
    }

    public static String orderUrl() {
        switch (env) {
            case "uat": return "https://uat-order.company.com";
            case "prod": return "https://order.company.com";
            default: return "https://qa-order.company.com";
        }
    }
}

package requests.auth;

import static io.gatling.javaapi.core.CoreDsl.*;
import static io.gatling.javaapi.http.HttpDsl.*;

public class SamlLoginRequest {

    public static ChainBuilder samlLogin =

        exec(
            http("Generate SAML")
                .post("/saml/generate")
                .body(StringBody("""
                {
                  "username": "#{username}",
                  "password": "#{password}"
                }
                """))
                .check(status().is(200))
                .check(jsonPath("$.samlResponse").saveAs("samlResponse"))
        )

        .exec(
            http("Exchange SAML for Token")
                .post("/oauth/token")
                .formParam("grant_type",
                    "urn:ietf:params:oauth:grant-type:saml2-bearer")
                .formParam("assertion", "#{samlResponse}")
                .check(status().is(200))
                .check(jsonPath("$.access_token").saveAs("authToken"))
        );
}

package simulations;

import io.gatling.javaapi.core.*;
import requests.auth.SamlLoginRequest;
import requests.user.GetUserRequest;
import requests.order.CreateOrderRequest;
import utils.BaseConfig;

import static io.gatling.javaapi.core.CoreDsl.*;

public class FullSimulation extends Simulation {

    FeederBuilder<String> userFeeder =
        csv("data/users.csv").circular();

    FeederBuilder<String> orderFeeder =
        csv("data/orders.csv").random();

    ScenarioBuilder scn = scenario("Full Enterprise Flow")
        .feed(userFeeder)
        .exec(SamlLoginRequest.samlLogin)
        .exec(GetUserRequest.getUser)
        .feed(orderFeeder)
        .exec(CreateOrderRequest.createOrder);

    {
        setUp(
            scn.injectClosed(

                rampConcurrentUsers(0)
                    .to(1000)
                    .during(20 * 60),

                constantConcurrentUsers(1000)
                    .during(60 * 60),

                rampConcurrentUsers(1000)
                    .to(0)
                    .during(20 * 60)

            )
        )
        .protocols(
            BaseConfig.authProtocol,
            BaseConfig.userProtocol,
            BaseConfig.orderProtocol
        )
        .assertions(
            global().responseTime().percentile(95).lt(2000),
            global().successfulRequests().percent().gt(99.0)
        );
    }
}
