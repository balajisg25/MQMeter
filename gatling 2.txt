    public static HttpProtocolBuilder yamProtocol =
        http.baseUrl("http://hello:1000")
            .acceptHeader("application/json");

    public static HttpProtocolBuilder balancesProtocol =
        http.baseUrl("https://website")
            .acceptHeader("application/json")
            .keyStore("/path/to/client.jks", "password"); // mTLS certificate
}

package requests.yam;

import static io.gatling.javaapi.core.CoreDsl.*;
import static io.gatling.javaapi.http.HttpDsl.*;

import io.gatling.javaapi.core.ChainBuilder;
import io.gatling.javaapi.http.HttpDsl;

public class YamGenerateRequest {

    public static ChainBuilder generateYam =
        exec(
            http("YAM Generate")
                .post("/yamVwww/generate")
                .header("Connection", "keep-alive")
                .header("Content-Type", "application/json")
                .body(ElFileBody("data/yam_payload.json"))
                // Extract custom header value, e.g., X-Custom-Token
                .check(header("X-Custom-Token").saveAs("customToken"))
                .check(status().is(200))
        )
        .pause(2); // think time
}

package requests.balances;

import static io.gatling.javaapi.core.CoreDsl.*;
import static io.gatling.javaapi.http.HttpDsl.*;

import io.gatling.javaapi.core.ChainBuilder;

public class GetBalancesRequest {

    public static ChainBuilder getBalances =
        exec(
            http("Get Account Balances")
                .get("/META-fa01/N1fcash-accl/balances/HE44001846748745/EUR/balances")
                .queryParam("datefrom", "2025-04-01")
                .queryParam("dateTo", "2026-02-19")
                .header("Accept", "application/json")
                // pass extracted header from POST
                .header("X-Custom-Token", "#{customToken}")
                .check(status().is(200))
        )
        .pause(2);
}

package simulations;

import io.gatling.javaapi.core.*;
import requests.yam.YamGenerateRequest;
import requests.balances.GetBalancesRequest;
import utils.BaseConfig;

import static io.gatling.javaapi.core.CoreDsl.*;
import static io.gatling.javaapi.http.HttpDsl.*;

public class FullSimulation extends Simulation {

    ScenarioBuilder scn = scenario("Yam + Balances Flow")
        .exec(YamGenerateRequest.generateYam)
        .exec(GetBalancesRequest.getBalances)
        .pace(5); // dynamic pacing, 5 sec

    {
        setUp(
            // 1 user for script validation
            scn.injectOpen(atOnceUsers(1))
        ).protocols(BaseConfig.yamProtocol, BaseConfig.balancesProtocol)
         .assertions(
             global().responseTime().percentile(95).lt(3000),
             global().successfulRequests().percent().gt(99)
         );

        // Load test scenario
        setUp(
            scn.injectOpen(
                rampUsers(100)
                    .increment(10)
                    .every(java.time.Duration.ofSeconds(30)),
                constantUsersPerSec(100)
                    .during(java.time.Duration.ofMinutes(30)),
                rampUsers(100)
                    .increment(-10)
                    .every(java.time.Duration.ofSeconds(30))
            )
        )
        .protocols(BaseConfig.yamProtocol, BaseConfig.balancesProtocol)
        .assertions(
            global().responseTime().percentile(95).lt(3000),
            global().successfulRequests().percent().gt(99)
        );
    }
}
