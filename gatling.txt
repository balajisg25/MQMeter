🔐 1️⃣ Login Request (Correlation Example)
📁 requests/LoginRequest.java
Java
Copy code
package requests;

import static io.gatling.javaapi.core.CoreDsl.*;
import static io.gatling.javaapi.http.HttpDsl.*;

public class LoginRequest {

    public static ChainBuilder login =
        exec(
            http("Login API")
                .post("/login")
                .body(StringBody(
                    """
                    {
                      "username": "#{username}",
                      "password": "#{password}"
                    }
                    """
                ))
                .check(status().is(200))
                // Correlation: extract token
                .check(jsonPath("$.token").saveAs("authToken"))
        );
}

✅ STEP 7 — Main Simulation
📁 simulations/ApiSimulation.java
Java
Copy code
package simulations;

import io.gatling.javaapi.core.*;
import io.gatling.javaapi.http.*;
import requests.*;
import utils.BaseConfig;

import static io.gatling.javaapi.core.CoreDsl.*;

public class ApiSimulation extends Simulation {

    FeederBuilder<String> userFeeder =
        csv("data/users.csv").circular();

    FeederBuilder<String> orderFeeder =
        csv("data/orders.csv").random();

    ScenarioBuilder scn = scenario("Full API Flow")
        .feed(userFeeder)
        .exec(LoginRequest.login)
        .feed(orderFeeder)
        .exec(GetUserRequest.getUser)
        .exec(CreateOrderRequest.createOrder);

    {
        setUp(
            scn.injectOpen(rampUsers(50).during(30))
        )
        .protocols(BaseConfig.httpProtocol)
        .assertions(
            global().responseTime().max().lt(2000),
            global().successfulRequests().percent().gt(95.0)
        );
    }
}
